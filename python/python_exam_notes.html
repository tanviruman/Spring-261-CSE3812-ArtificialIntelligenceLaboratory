Class structure — MEMORIZE THIS

class BankAccount:
    count = 0  # class/static variable (shared)

    def __init__(self, account_number, balance=0):
        self.account_number = account_number  # instance var
        self.balance = balance
        BankAccount.count += 1

    def deposit(self, amount):
        self.balance += amount

    def withdraw(self, amount):
        if amount <= self.balance:
            self.balance -= amount
        else:
            print("Insufficient funds")

    def display(self):
        print(f"Acc: {self.account_number}, Bal: {self.balance}")

    def __str__(self):       # called by print()
        return f"Account {self.account_number}"

    @staticmethod
    def is_valid(amount):    # no self needed
        return amount > 0

# Create object and test
acc = BankAccount("001", 1000)
acc.deposit(500)
acc.withdraw(200)
acc.display()

Key OOP vocabulary

__init__   → constructor
self       → like "this" in Java/C++
__str__    → toString() — called by print()
@staticmethod → same for all objects, no self
class var  → shared across all objects
instance var → unique to each object
Python does NOT support overloading

If you define two functions with same name, the second one overwrites the first.
